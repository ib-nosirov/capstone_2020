	\frame {
		\frametitle{Problem 2: Solving Exponential Equations}
		Solve for x and round to the nearest hundredth
		\begin{align*}
			8*8^{10x} = 24
		\end{align*}
	}

	\frame {
		\frametitle{Problem 2: Solving Exponential Equations}
		Solve for x and round to the nearest hundredth
		\begin{align*}
			8*8^{10x} = 24\\  
		  ->	8^{10x} = \dfrac{\color{blue}24}{\color{blue}8}\\
		\end{align*}
	}

	\frame {
		\frametitle{Problem 2: Solving Exponential Equations}
		Solve for x and round to the nearest hundredth
		\begin{align*}
			8^{10x} = 3\\
			-> 	\color{blue}log \color{black}8^{10x} = \color{blue}log \color{black}3\\
		\end{align*}
	}

	\frame {
		\frametitle{Problem 2: Solving Exponential Equations}
		Solve for x and round to the nearest hundredth
		\begin{align*}
			\color{black}log8^{10x} = log3\\
		-> \color{blue}10x \color{black}log8 = log3\\
		\end{align*}
	}

	\frame {
		\frametitle{Problem 2: Solving Exponential Equations}
		Solve for x and round to the nearest hundredth
		\begin{align*}
			10x*log8 = log3\\
		-> 10x = \dfrac{log3}{\color{blue}log8}\\
		\end{align*}
	}

	\frame {
		\frametitle{Problem 2: Solving Exponential Equations}
		Solve for x and round to the nearest hundredth
		\begin{align*}
			10x = 0.5283208\\
			x = \dfrac{0.5283208}{\color{blue}10}\\
			x = 0.053
		\end{align*}
	}
